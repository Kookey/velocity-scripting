53    public boolean render(final Params params, final Writer writer) throws IOException, ResourceNotFoundException, ParseErrorException, MethodInvocationException {
54      int leftIndex = 0;
55      int rightIndex = params.maxBody;
56      if (!params.prefixOverrides.isEmpty()) {
57        final String LEFT = params.body.substring(0, params.maxPrefixLength < params.maxBody ? params.maxPrefixLength : params.maxBody).toUpperCase(Locale.ENGLISH);
58        FastLinkedList<String>.Node n = params.prefixOverrides.start();
59        while (n != null) {
60          if (LEFT.startsWith(n.data)) {
61            leftIndex = n.data.length();
62            break;
63          }
64          n = n.next;
65        }
66      }
67      if (!params.suffixOverrides.isEmpty()) {
68        final String RIGHT = params.body.substring(rightIndex - params.maxSuffixLength).toUpperCase(Locale.ENGLISH);
69        FastLinkedList<String>.Node n = params.suffixOverrides.start();
70        while (n != null) {
71          if (RIGHT.endsWith(n.data)) {
72            rightIndex = rightIndex - n.data.length();
73            break;
74          }
75          n = n.next;
76        }
77      }
78      writer.append(params.prefix).append(' ');
79      writer.append(params.body, leftIndex, rightIndex).append(' ');
80      writer.append(params.suffix);
81      return true;
82    }

143   protected Params getParams(final InternalContextAdapter context, final Node node) throws IOException, ResourceNotFoundException, ParseErrorException, MethodInvocationException {
144     final Params params = new Params();
145     final int nodes = node.jjtGetNumChildren();
146     for (int i = 0; i < nodes; i++) {
147       Node child = node.jjtGetChild(i);
148       if (child != null) {
149         if (!(child instanceof ASTBlock)) {
150           if (i == 0) {
151             params.setPrefix(String.valueOf(child.value(context)));
152           } else if (i == 1) {
153             params.setPrefixOverrides(String.valueOf(child.value(context)).toUpperCase(Locale.ENGLISH));
154           } else if (i == 2) {
155             params.setSuffix(String.valueOf(child.value(context)));
156           } else if (i == 3) {
157             params.setSuffixOverrides(String.valueOf(child.value(context)).toUpperCase(Locale.ENGLISH));
158           } else {
159             break;
160           }
161         } else {
162           StringWriter blockContent = new StringWriter();
163           child.render(context, blockContent);
164           params.setBody(blockContent.toString().trim());
165           break;
166         }
167       }
168     }
169     return params;
170   }

172   private static int fromStringList(final String list, final char sep, final FastLinkedList<String> fll) {
173     int max = 0;
174     if (list != null) {
175       final int n = list.length();
176       int i = 0;
177       while (i < n) {
178         int r = list.indexOf(sep, i);
179         if (i < r) {
180           fll.add(list.substring(i, r));
181           int len = r - i;
182           if (len > max) {
183             max = len;
184           }
185           i = r+1;
186         }
187         else {
188           break;
189         }
190       }
191       if (i < n) {
192         fll.add(list.substring(i));
193         int len = n - i;
194         if (len > max) {
195           max = len;
196         }
197       }
198     }
199     return max;
200   }
